<html lang="he" dir="rtl">

<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>אפליקציית סקר רב-שלבית</title>

    <style>

        body {

            font-family: Arial, sans-serif;

            max-width: 600px;

            margin: 0 auto;

            padding: 20px;

        }

        .card {

            border: 1px solid #ddd;

            border-radius: 8px;

            padding: 20px;

            margin-bottom: 20px;

        }

        input, select, button {

            width: 100%;

            padding: 10px;

            margin: 10px 0;

        }

        button {

            background-color: #007bff;

            color: white;

            border: none;

            cursor: pointer;

        }

        button:hover {

            background-color: #0056b3;

        }

    </style>

</head>

<body>

    <div id="app"></div>

 

    <script>

        const categories = {

            fruits: 'פירות',

            vegetables: 'ירקות',

            sweets: 'ממתקים',

            carbs: 'פחמימות'

        };

 

        const questions = {

            fruits: ['מה הפרי האהוב עליך?', 'מה הפרי השנוא עליך?'],

            vegetables: ['מה הירק האהוב עליך?', 'מה הירק השנוא עליך?'],

            sweets: ['מה הממתק האהוב עליך?', 'מה הממתק השנוא עליך?'],

            carbs: ['מה הפחמימה האהובה עליך?', 'מה הפחמימה השנואה עליך?']

        };

 

        let step = 0;

        let name = '';

        let category = '';

        let answers = ['', ''];

        let responses = [];

 

        function renderApp() {

            const app = document.getElementById('app');

            app.innerHTML = '';

 

            switch (step) {

                case 0:

                    app.innerHTML = `

                        <div class="card">

                            <h2>מה שמך?</h2>

                            <input type="text" id="nameInput" placeholder="הכנס את שמך" value="${name}">

                            <button onclick="nextStep()">הבא</button>

                        </div>

                    `;

                    break;

                case 1:

                    app.innerHTML = `

                        <div class="card">

                            <h2>בחר קטגוריה</h2>

                            <select id="categorySelect">

                                <option value="">בחר קטגוריה</option>

                                ${Object.entries(categories).map(([key, value]) => `<option value="${key}">${value}</option>`).join('')}

                            </select>

                            <button onclick="nextStep()">הבא</button>

                        </div>

                    `;

                    break;

                case 2:

                    app.innerHTML = `

                        <div class="card">

                            <h2>שאלון ${categories[category]}</h2>

                            ${questions[category].map((question, index) => `

                                <div>

                                    <label>${question}</label>

                                    <input type="text" id="answer${index}" placeholder="הכנס את תשובתך" value="${answers[index]}">

                                </div>

                            `).join('')}

                            <button onclick="submitAnswers()">שלח</button>

                        </div>

                    `;

                    break;

            }

 

            renderStatistics();

        }

 

        function nextStep() {

            if (step === 0) {

                name = document.getElementById('nameInput').value;

                if (!name) {

                    alert('נא להזין שם');

                    return;

                }

            } else if (step === 1) {

                category = document.getElementById('categorySelect').value;

                if (!category) {

                    alert('נא לבחור קטגוריה');

                    return;

                }

            }

            step++;

            renderApp();

        }

 

        function submitAnswers() {

            answers = [

                document.getElementById('answer0').value,

                document.getElementById('answer1').value

            ];

            if (!answers[0] || !answers[1]) {

                alert('נא לענות על כל השאלות');

                return;

            }

            responses.push({ name, category, answers });

            step = 0;

            name = '';

            category = '';

            answers = ['', ''];

            renderApp();

        }

 

        function renderStatistics() {

            if (responses.length === 0) return;

 

            const stats = Object.keys(categories).reduce((acc, cat) => {

                acc[cat] = { count: 0, favorites: {}, hated: {} };

                return acc;

            }, {});

 

            responses.forEach(({ category, answers }) => {

                stats[category].count++;

                answers.forEach((answer, index) => {

                    const key = index === 0 ? 'favorites' : 'hated';

                    stats[category][key][answer] = (stats[category][key][answer] || 0) + 1;

                });

            });

 

            const statsHtml = Object.entries(stats).map(([category, data]) => `

                <div>

                    <h3>${categories[category]} (${data.count} תגובות)</h3>

                    <p>מועדפים: ${Object.entries(data.favorites).sort((a, b) => b[1] - a[1]).map(([item, count]) => `${item} (${count})`).join(', ')}</p>

                    <p>שנואים: ${Object.entries(data.hated).sort((a, b) => b[1] - a[1]).map(([item, count]) => `${item} (${count})`).join(', ')}</p>

                </div>

            `).join('');

 

            const statsElement = document.createElement('div');

            statsElement.className = 'card';

            statsElement.innerHTML = `

                <h2>סטטיסטיקות</h2>

                ${statsHtml}

            `;

            document.getElementById('app').appendChild(statsElement);

        }

 

        renderApp();

    </script>

</body>

</html>
